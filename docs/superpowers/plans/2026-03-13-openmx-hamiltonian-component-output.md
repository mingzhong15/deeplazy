/**********************************************************************
  Count_H_Blocks: Count total number of atom-neighbor blocks
***********************************************************************/
static int Count_H_Blocks(void)
{
    int Gc_AN, Mc_AN, h_AN;
    int count = 0;

    for (Mc_AN = 1; Mc_AN <= Matomnum; Mc_AN++) {
        Gc_AN = M2G[Mc_AN];
        for (h_AN = 0; h_AN <= FNAN[Gc_AN]; h_AN++) {
            count++;
        }
    }
    return count;
}

/**********************************************************************
  Pack_Hamiltonian_Data: Pack Hamiltonian data into contiguous buffers

  For each atom-neighbor block, store:
    - Matrix elements: data[block_idx][orb_i][orb_j]
    - Indices: indices[block_idx][3] = (center_atom, neighbor_atom, cell_id)
    - Orbital sizes: orb_sizes[block_idx][2] = (n_orb_center, n_orb_neighbor)
***********************************************************************/
static void Pack_Hamiltonian_Data(
    double *data_buffer,
    int *indices_buffer,
    int *orb_sizes_buffer,
    int spin,
    int is_H0,        /* 1 for H0 (spin-independent), 0 otherwise */
    int is_HVNA,      /* 1 for HVNA (spin-independent), 0 otherwise */
    int is_OLP,       /* 1 for overlap matrix, 0 otherwise */
    int *total_blocks,
    int *total_elements)
{
    int Gc_AN, Mc_AN, h_AN, Gh_AN, Rn;
    int Cwan, Hwan, tno0, tno1;
    int i, j;
    int block_idx = 0;
    int data_idx = 0;
    int spin_idx;

    /* Determine spin index */
    if (is_H0 || is_HVNA || is_OLP) {
        spin_idx = 0;  /* Spin-independent */
    } else {
        spin_idx = spin;
    }

    for (Mc_AN = 1; Mc_AN <= Matomnum; Mc_AN++) {
        Gc_AN = M2G[Mc_AN];
        Cwan = WhatSpecies[Gc_AN];
        tno0 = Spe_Total_CNO[Cwan];

        for (h_AN = 0; h_AN <= FNAN[Gc_AN]; h_AN++) {
            Gh_AN = natn[Gc_AN][h_AN];
            Rn = ncn[Gc_AN][h_AN];
            Hwan = WhatSpecies[Gh_AN];
            tno1 = Spe_Total_CNO[Hwan];

            /* Store indices */
            indices_buffer[block_idx * 3 + 0] = Gc_AN;
            indices_buffer[block_idx * 3 + 1] = Gh_AN;
            indices_buffer[block_idx * 3 + 2] = Rn;

            /* Store orbital sizes */
            orb_sizes_buffer[block_idx * 2 + 0] = tno0;
            orb_sizes_buffer[block_idx * 2 + 1] = tno1;

            /* Store matrix elements */
            if (is_H0) {
                for (i = 0; i < tno0; i++) {
                    for (j = 0; j < tno1; j++) {
                        data_buffer[data_idx++] = H0[0][Mc_AN][h_AN][i][j];
                    }
                }
            } else if (is_HVNA) {
                for (i = 0; i < tno0; i++) {
                    for (j = 0; j < tno1; j++) {
                        data_buffer[data_idx++] = HVNA[Mc_AN][h_AN][i][j];
                    }
                }
            } else if (is_OLP) {
                for (i = 0; i < tno0; i++) {
                    for (j = 0; j < tno1; j++) {
                        data_buffer[data_idx++] = OLP[0][Mc_AN][h_AN][i][j];
                    }
                }
            } else {
                for (i = 0; i < tno0; i++) {
                    for (j = 0; j < tno1; j++) {
                        data_buffer[data_idx++] = H[spin][Mc_AN][h_AN][i][j];
                    }
                }
            }

            block_idx++;
        }
    }

    *total_blocks = block_idx;
    *total_elements = data_idx;
}

/**********************************************************************
  Write_Hamiltonian_HDF5: Write Hamiltonian data to HDF5 file

  File structure:
    /matrix/data           - float64 array of matrix elements
    /matrix/indices        - int32 array of (center_atom, neighbor_atom, cell)
    /matrix/orbital_sizes  - int32 array of (n_orb_i, n_orb_j) per block
    /metadata/num_spins    - int32 scalar
    /metadata/num_blocks   - int32 scalar
***********************************************************************/
static void Write_Hamiltonian_HDF5(
    const char *output_filename,
    double *data,
    int *indices,
    int *orb_sizes,
    int num_blocks,
    int num_elements,
    int num_spins)
{
    hid_t file_id, group_id;
    hid_t dataspace_id, dataset_id;
    hid_t plist_id;
    hsize_t dims[1];
    herr_t status;
    char full_path[YOUSO10];

    /* Create full path */
    sprintf(full_path, "%s%s", filepath, output_filename);

    /* Create HDF5 file */
    file_id = H5Fcreate(full_path, H5F_ACC_TRUNC, H5P_DEFAULT, H5P_DEFAULT);
    if (file_id < 0) {
        printf("Error: Cannot create HDF5 file %s\n", full_path);
        return;
    }

    /* Create /matrix group */
    group_id = H5Gcreate2(file_id, "/matrix", H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);

    /* Write data array */
    dims[0] = num_elements;
    dataspace_id = H5Screate_simple(1, dims, NULL);
    plist_id = H5Pcreate(H5P_DATASET_CREATE);
    H5Pset_chunk(plist_id, 1, dims);
    H5Pset_deflate(plist_id, 4);  /* gzip compression level 4 */
    dataset_id = H5Dcreate2(group_id, "data", H5T_IEEE_F64LE, dataspace_id,
                            H5P_DEFAULT, plist_id, H5P_DEFAULT);
    H5Dwrite(dataset_id, H5T_NATIVE_DOUBLE, H5S_ALL, H5S_ALL, H5P_DEFAULT, data);
    H5Dclose(dataset_id);
    H5Pclose(plist_id);
    H5Sclose(dataspace_id);

    /* Write indices array */
    dims[0] = num_blocks * 3;
    dataspace_id = H5Screate_simple(1, dims, NULL);
    dataset_id = H5Dcreate2(group_id, "indices", H5T_STD_I32LE, dataspace_id,
                            H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
    H5Dwrite(dataset_id, H5T_NATIVE_INT, H5S_ALL, H5S_ALL, H5P_DEFAULT, indices);
    H5Dclose(dataset_id);
    H5Sclose(dataspace_id);

    /* Write orbital_sizes array */
    dims[0] = num_blocks * 2;
    dataspace_id = H5Screate_simple(1, dims, NULL);
    dataset_id = H5Dcreate2(group_id, "orbital_sizes", H5T_STD_I32LE, dataspace_id,
                            H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
    H5Dwrite(dataset_id, H5T_NATIVE_INT, H5S_ALL, H5S_ALL, H5P_DEFAULT, orb_sizes);
    H5Dclose(dataset_id);
    H5Sclose(dataspace_id);

    H5Gclose(group_id);

    /* Create /metadata group */
    group_id = H5Gcreate2(file_id, "/metadata", H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);

    /* Write num_spins */
    dims[0] = 1;
    dataspace_id = H5Screate_simple(1, dims, NULL);
    dataset_id = H5Dcreate2(group_id, "num_spins", H5T_STD_I32LE, dataspace_id,
                            H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
    H5Dwrite(dataset_id, H5T_NATIVE_INT, H5S_ALL, H5S_ALL, H5P_DEFAULT, &num_spins);
    H5Dclose(dataset_id);
    H5Sclose(dataspace_id);

    /* Write num_blocks */
    dataset_id = H5Dcreate2(group_id, "num_blocks", H5T_STD_I32LE, dataspace_id,
                            H5P_DEFAULT, H5P_DEFAULT, H5P_DEFAULT);
    H5Dwrite(dataset_id, H5T_NATIVE_INT, H5S_ALL, H5S_ALL, H5P_DEFAULT, &num_blocks);
    H5Dclose(dataset_id);
    H5Sclose(dataspace_id);

    H5Gclose(group_id);
    H5Fclose(file_id);

    printf("  Written: %s (%d blocks, %d elements)\n", full_path, num_blocks, num_elements);
}

/**********************************************************************
  Output_H_Total_HDF5: Output total Hamiltonian to H_total.h5
***********************************************************************/
void Output_H_Total_HDF5(void)
{
    int spin;
    int num_blocks, num_elements;
    int total_blocks;
    int max_elements;
    double *data_buffer;
    int *indices_buffer;
    int *orb_sizes_buffer;

    /* Count blocks and estimate buffer size */
    total_blocks = Count_H_Blocks();

    /* Allocate buffers (conservative estimate) */
    max_elements = total_blocks * List_YOUSO[7] * List_YOUSO[7];
    data_buffer = (double *)malloc(sizeof(double) * max_elements);
    indices_buffer = (int *)malloc(sizeof(int) * total_blocks * 3);
    orb_sizes_buffer = (int *)malloc(sizeof(int) * total_blocks * 2);

    /* Output for each spin channel */
    for (spin = 0; spin <= SpinP_switch; spin++) {
        char filename[64];
        sprintf(filename, "H_total_spin%d.h5", spin);

        Pack_Hamiltonian_Data(data_buffer, indices_buffer, orb_sizes_buffer,
                              spin, 0, 0, 0, &num_blocks, &num_elements);

        Write_Hamiltonian_HDF5(filename, data_buffer, indices_buffer, orb_sizes_buffer,
                               num_blocks, num_elements, SpinP_switch + 1);
    }

    free(data_buffer);
    free(indices_buffer);
    free(orb_sizes_buffer);
}

"""
OpenMX Hamiltonian Component Reader

Reads HDF5 files output by OpenMX with Hamiltonian.Components.Output enabled.
Computes SCF-dependent Hamiltonian component for machine learning.
"""

import h5py
import numpy as np
from pathlib import Path
from typing import Dict, Optional, Tuple


class OpenMXHamiltonianReader:
    """
    Reader for OpenMX Hamiltonian component HDF5 files.
    
    Usage:
        reader = OpenMXHamiltonianReader('/path/to/output/')
        H_scf = reader.compute_H_SCF()
    """
    
    def __init__(self, directory: str):
        """
        Initialize reader with output directory.
        
        Args:
            directory: Path to directory containing HDF5 files
        """
        self.directory = Path(directory)
        self._cache = {}
    
    def load_hdf5(self, filename: str) -> Dict:
        """Load an HDF5 file into a dictionary."""
        filepath = self.directory / filename
        if not filepath.exists():
            raise FileNotFoundError(f"File not found: {filepath}")
        
        if filename not in self._cache:
            data = {}
            with h5py.File(filepath, 'r') as f:
                # Load matrix data
                if 'matrix' in f:
                    grp = f['matrix']
                    data['data'] = grp['data'][:]
                    data['indices'] = grp['indices'][:]
                    data['orbital_sizes'] = grp['orbital_sizes'][:]
                
                # Load metadata
                if 'metadata' in f:
                    grp = f['metadata']
                    data['num_spins'] = int(grp['num_spins'][()])
                    if 'num_blocks' in grp:
                        data['num_blocks'] = int(grp['num_blocks'][()])
            
            self._cache[filename] = data
        
        return self._cache[filename]
    
    def load_structure(self) -> Dict:
        """Load structure information."""
        data = self.load_hdf5('structure.h5')
        
        with h5py.File(self.directory / 'structure.h5', 'r') as f:
            if 'structure' in f:
                grp = f['structure']
                data['atomic_positions'] = grp['atomic_positions'][:]
                data['lattice_vectors'] = grp['lattice_vectors'][:]
                data['atomic_species'] = grp['atomic_species'][:]
                data['orbital_counts'] = grp['orbital_counts'][:]
        
        return data
    
    def load_H_total(self, spin: int = 0) -> Dict:
        """Load total Hamiltonian for a given spin channel."""
        filename = f'H_total_spin{spin}.h5'
        return self.load_hdf5(filename)
    
    def load_H0(self) -> Dict:
        """Load kinetic energy Hamiltonian."""
        return self.load_hdf5('H0.h5')
    
    def load_HNL(self, spin: int = 0) -> Dict:
        """Load nonlocal pseudopotential for a given spin channel."""
        filename = f'HNL_spin{spin}.h5'
        return self.load_hdf5(filename)
    
    def load_iHNL(self, spin: int = 0) -> Optional[Dict]:
        """Load imaginary nonlocal part (SOC)."""
        filename = f'iHNL_spin{spin}.h5'
        filepath = self.directory / filename
        if filepath.exists():
            return self.load_hdf5(filename)
        return None
    
    def load_HVNA(self) -> Dict:
        """Load VNA potential."""
        return self.load_hdf5('HVNA.h5')
    
    def load_S(self) -> Dict:
        """Load overlap matrix."""
        return self.load_hdf5('S.h5')
    
    def reconstruct_sparse_matrix(self, data: Dict) -> Tuple[np.ndarray, np.ndarray, np.ndarray]:
        """
        Reconstruct sparse matrix from packed format.
        
        Returns:
            Tuple of (data_array, indices_array, orbital_sizes)
        """
        return data['data'], data['indices'], data['orbital_sizes']
    
    def compute_H_SCF(self, spin: int = 0) -> Dict:
        """
        Compute SCF-dependent Hamiltonian component.
        
        H_SCF = H_total - H0 - HNL - iHNL*i - HVNA
        
        Args:
            spin: Spin channel index (default 0)
        
        Returns:
            Dictionary with computed H_SCF and metadata
        """
        # Load all components
        H_total = self.load_H_total(spin)
        H0 = self.load_H0()
        HNL = self.load_HNL(spin)
        HVNA = self.load_HVNA()
        iHNL = self.load_iHNL(spin)
        
        # Compute H_SCF
        H_scf_data = H_total['data'] - H0['data'] - HNL['data'] - HVNA['data']
        
        # Add imaginary part for SOC
        if iHNL is not None:
            # For complex matrices, this would need to handle real and imaginary separately
            # Here we just subtract the real part
            pass
        
        result = {
            'data': H_scf_data,
            'indices': H_total['indices'],
            'orbital_sizes': H_total['orbital_sizes'],
            'num_spins': H_total['num_spins'],
        }
        
        return result
    
    def get_summary(self) -> str:
        """Get a summary of loaded data."""
        structure = self.load_structure()
        summary = [
            "OpenMX Hamiltonian Data Summary",
            "=" * 40,
            f"Directory: {self.directory}",
            f"Number of atoms: {structure.get('num_atoms', 'N/A')}",
            f"Number of spins: {structure.get('num_spins', 'N/A')}",
            f"Fermi level: {structure.get('fermi_level', 'N/A')} Ha",
        ]
        return "\n".join(summary)


def read_openmx_hamiltonian(directory: str) -> OpenMXHamiltonianReader:
    """
    Convenience function to create a reader.
    
    Args:
        directory: Path to output directory
    
    Returns:
        OpenMXHamiltonianReader instance
    """
    return OpenMXHamiltonianReader(directory)
